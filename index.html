async function handleGemini(e) { if (e.key === 'Enter') { const input = document.getElementById('user-input'); const chat = document.getElementById('chat-area'); const prompt = input.value.trim(); if (!prompt) return; chat.innerHTML += `
${prompt}
`; input.value = ""; chat.scrollTop = chat.scrollHeight; const tempMsg = document.createElement("div"); tempMsg.className = "msg g-msg"; tempMsg.innerText = "Düşünüyorum..."; chat.appendChild(tempMsg); try { const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${API_KEY}`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }] }) }); const data = await response.json(); // Hata mesajını daha detaylı görmek için: if (data.error) { tempMsg.innerText = "Google Hatası: " + data.error.message; return; } const result = data.candidates[0].content.parts[0].text; tempMsg.innerText = result; } catch (error) { tempMsg.innerText = "Bağlantı koptu! İnternetini veya API anahtarını kontrol et Hamza."; console.error(error); } chat.scrollTop = chat.scrollHeight; } }
